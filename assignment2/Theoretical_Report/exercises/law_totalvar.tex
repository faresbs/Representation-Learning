\Exercise{
\label{ex:law_totalvar}
Recall the variance of $X$ is $\Var(X)=\E[(X-\E[X])^2]$.
\begin{enumerate}
\item Let $X$ be a random variable with finite mean. 
Show $\Var(X) = \E[X^2] - \E[X]^2$.

\item Let $X$ and $Z$ be random variables on the same probability space.
Show that $\Var(X) = \E_Z[\Var(X|Z)] + \Var_Z(\E[X|Z])$.
(Hint: $\E[X]=\E_Y[\E[X|Y]]$.)
\end{enumerate}
